Bash

npm install -g prettyaml
# From file
prettyaml example.json

# From stdin
cat example.json | prettyaml
  • Install NodeJS
  • prettyaml

JavaScript

(function () {
  "use strict";
  
  var YAML = require('prettyaml')
    , json
    , data
    , yml
    ;
    
  json = '{ "foo": "bar" }';
  data = JSON.parse(json);
  yml = prettyaml.stringify(data);
  
  console.log(yml);
}());
  • npm install prettyaml
  • JSON.parse
  • prettyaml.stringify

Python

import json
import pyaml

str = '{ "foo": "bar" }'
data = json.loads(str)
yml = pyaml.dump(data)

print(yml)
  • pip install pyaml
  • json.loads
  • pyaml.dump
